- [ ] **Step 2: shadcn primary 색상을 카키로 재정의**

  `app/globals.css`의 `:root` 블록(shadcn이 생성한 부분)을 찾아 `--primary` 값을 수정. shadcn은 HSL 형식을 사용.
  `#8B8378` = `hsl(35 8% 51%)`.

  ```css
  :root {
    /* 기존 shadcn 변수 중 아래 두 줄만 수정 */
    --primary: 35 8% 51%;
    --primary-foreground: 0 0% 100%;
    /* 나머지 shadcn 변수는 그대로 유지 */
  }
  ```

  > 참고: `:root` 블록이 없으면 shadcn init이 생성하지 않은 것이므로 이 단계 건너뜀.

## Task 6: 온보딩 Context

**Files:** `lib/onboarding/context.tsx`

- [ ] **Step 1: `lib/onboarding/context.tsx` 작성**

  ```typescript
  'use client'

  import { createContext, useContext, useReducer, type ReactNode } from 'react'
  import type {
    OnboardingState,
    OnboardingAction,
    OnboardingStep,
  } from '@/lib/types/onboarding'

  const initialState: OnboardingState = {
    currentStep: 1,
    formData: {},
    isComplete: false,
  }

  function onboardingReducer(
    state: OnboardingState,
    action: OnboardingAction
  ): OnboardingState {
    switch (action.type) {
      case 'SET_STEP':
        return { ...state, currentStep: action.payload }
      case 'UPDATE_FORM':
        return { ...state, formData: { ...state.formData, ...action.payload } }
      case 'COMPLETE':
        return { ...state, isComplete: true }
      case 'RESET':
        return initialState
      default:
        return state
    }
  }

  interface OnboardingContextValue {
    state: OnboardingState
    setStep: (step: OnboardingStep) => void
    updateForm: (data: Partial<OnboardingState['formData']>) => void
    complete: () => void
    reset: () => void
  }

  const OnboardingContext = createContext<OnboardingContextValue | null>(null)

  export function OnboardingProvider({ children }: { children: ReactNode }) {
    const [state, dispatch] = useReducer(onboardingReducer, initialState)

    return (
      <OnboardingContext.Provider
        value={{
          state,
          setStep: (step) => dispatch({ type: 'SET_STEP', payload: step }),
          updateForm: (data) => dispatch({ type: 'UPDATE_FORM', payload: data }),
          complete: () => dispatch({ type: 'COMPLETE' }),
          reset: () => dispatch({ type: 'RESET' }),
        }}
      >
        {children}
      </OnboardingContext.Provider>
    )
  }

  export function useOnboarding(): OnboardingContextValue {
    const ctx = useContext(OnboardingContext)
    if (!ctx) throw new Error('useOnboarding must be used within OnboardingProvider')
    return ctx
  }
  ```

                        ✕
                      </button>
                    </div>
                  </div>
                ))}
            </div>

          {grandTotal > 0 && (
            <div className="bg-[#F2EFE9] border border-[#D0CCC6] rounded-lg p-3 mb-4 text-sm">

            </button>
          </div>
        </div>
      </>
    )
  }

  ```typescript
  'use client'

  import { useState } from 'react'
  import ProgressSummary from '@/components/dashboard/ProgressSummary'
  import ChecklistCard from '@/components/dashboard/ChecklistCard'
  import { dummyChecklist, DUMMY_DEATH_DATE, DUMMY_DECEASED_NAME } from '@/lib/dummy/checklist'
  import type { ChecklistItem } from '@/lib/types/checklist'

  const groupByInstitution = (items: ChecklistItem[]) => {
    return items.reduce<Record<string, ChecklistItem[]>>((acc, item) => {
      const key = item.institution
      if (!acc[key]) acc[key] = []
      acc[key].push(item)
      return acc
    }, {})
  }

  export default function DashboardPage() {
    const [items, setItems] = useState<ChecklistItem[]>(dummyChecklist)

    const toggle = (id: string) => {
      setItems((prev) =>
        prev.map((item) =>
          item.id === id
            ? { ...item, status: item.status === 'completed' ? 'pending' : 'completed' }
            : item
        )
      )
    }

    const completed = items.filter((i) => i.status === 'completed').length
    const urgent = items.find((i) => i.status === 'urgent')
    const groups = groupByInstitution(items)

    return (
      <>
        <ProgressSummary
          deceasedName={DUMMY_DECEASED_NAME}
          deathDate={DUMMY_DEATH_DATE}
          completedCount={completed}
          totalCount={items.length}
          nextUrgentTitle={urgent?.title ?? '없음'}
          nextUrgentDue={urgent ? `${urgent.dueLabel} 기한` : ''}
        />

        <div className="space-y-6">
          {Object.entries(groups).map(([institution, groupItems]) => (
            <div key={institution}>
              <h3 className="text-xs font-bold text-[#8B8378] uppercase tracking-wide mb-2 px-1">
                {institution}
              </h3>
              <div className="space-y-2">
                {groupItems.map((item) => (
                  <ChecklistCard key={item.id} item={item} onToggle={toggle} />
                ))}
              </div>
            </div>
          ))}
        </div>
      </>
    )
  }
  ```
